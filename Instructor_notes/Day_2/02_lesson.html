class User(object):

    def __init__(self):
        print("Provide first name: ")
        self.name = input()
        print("Provide last name: ")
        self.last = input()

    def __repr__(self):
        return '<User %r>' % self.name

    def __str__(self):
        return self.name
    
    def change_first_name(self):
        print("Modify the first name")
        self.name = input()
    
    def change_last_name(self):
        print("Modify the last name")
        self.last = input()
